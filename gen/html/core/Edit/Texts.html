Texts Module¶

About¶

Command Guide¶

Module Texts defines the ‘text’ abstract data type used pervasively in the Oberon system.

API Guide¶

Dependencies¶

Files Fonts MIME

Constants¶

(*scanner symbol classes*)
   Inval* = 0;         (*invalid symbol*)
   Name* = 1;          (*name s (length len)*)
   String* = 2;        (*literal string s (length len)*)
   Int* = 3;           (*integer i (decimal or hexadecimal)*)
   Real* = 4;          (*real number x*)
   Char* = 6;          (*special character c*)

   (* TextBlock = TextTag offset run {run} "0" len {AsciiCode}.
     run = fnt [name] col voff len. *)

   TAB = 9X; CR = 0DX; NL = 0AX;
   TextTag = 0F1X;
   replace* = 0; insert* = 1; delete* = 2; unmark* = 3;  (*op-codes*)

Types¶

Piece = POINTER TO PieceDesc;
   PieceDesc = RECORD
     f: Files.File;
     off, len: LONGINT;
     fnt: Fonts.Font;
     col, voff: INTEGER;
     prev, next: Piece
   END;

   Text* = POINTER TO TextDesc;
   Buffer* = POINTER TO BufDesc;

   Processor* = PROCEDURE (B: Buffer): Buffer;
(* Feeder* = PROCEDURE(T: Text; R: Reader); *)

Notifier* = PROCEDURE (T: Text; op: INTEGER; beg, end: LONGINT);

TextDesc* = RECORD

len*,sent*,lcap*,bcap*: LONGINT; mime*: MIME.Type; changed*: BOOLEAN; notify*: Notifier; aproc*: Processor; trailer: Piece; pce: Piece; (cache) org: LONGINT (cache)

END;

Reader* = RECORD

eot*: BOOLEAN; fnt*: Fonts.Font; col*, voff*: INTEGER;

(* more*: Feeder; *)

ref: Piece; org: LONGINT; off: LONGINT; rider: Files.Rider

END;

Scanner* = RECORD (Reader)

nextCh*: CHAR; line*, class*: INTEGER; i*: LONGINT; x*: REAL; y*: LONGREAL; codepoint*: INTEGER; len*: INTEGER; s*: ARRAY 32 OF CHAR

END;

BufDesc* = RECORD

len*: LONGINT; header, last: Piece

END;

Writer* = RECORD

buf*: Buffer; fnt*: Fonts.Font; col*, voff*: INTEGER; rider: Files.Rider

END;

Stream* = RECORD

in*: Reader; out*: Writer

END;
